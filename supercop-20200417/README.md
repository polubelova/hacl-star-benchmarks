Script to print a table.

```
#!/bin/bash

for file in $( ls data )
do
 egrep -o " ok [0-9]+ [0-9]+ [0-9]+ [^ ]+ [^_]+" "$file" |
     sort -n -k2 | awk '{printf("%s %s %0.2f\n", $5, $6, ($2/16384))}' | awk '!a[$1]++' > "${file%}.txt"
done
```
